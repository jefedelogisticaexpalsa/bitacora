4. Sube index.html a GitHub → Settings → Pages → Branch: main → Save.
📋 Código para Google Apps Script 
function doPost(e) {
  var sheet = SpreadsheetApp
    .openById('1Aaj7t2bNjOfuvHazjxI6DLXN6cHUEAFc')
    .getActiveSheet();
  var data = JSON.parse(e.postData.contents);
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['FECHA','HORA','ENTRADA/SALIDA','GUARDIA','COMPAÑÍA',
      'NOMBRE/APELLIDO','PLACA','DISCO','KILOMETRAJE','COMBUSTIBLE',
      'RETORNO/DESTINO','LLANTA EMERGENCIA','EXTINTOR','KIT',
      'BINES','SACOS','GAVETAS','OBSERVACIONES']);
  }
  sheet.appendRow([data.fecha,data.hora,data.tipo,data.guardia,data.compania,
    data.nombre,data.placa,data.disco,data.kilometraje,data.combustible,
    data.destino,data.llanta,data.extintor,data.kit,
    data.bines,data.sacos,data.gavetas,data.observaciones]);
  return ContentService
    .createTextOutput(JSON.stringify({status:'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}
function doGet(e) {
  var sheet = SpreadsheetApp
    .openById('1Aaj7t2bNjOfuvHazjxI6DLXN6cHUEAFc')
    .getActiveSheet();
  var data = sheet.getDataRange().getValues();
  var output = ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
  return output;
}

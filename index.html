URL del Google Apps Script Web App
Pega aquí la URL de tu Web App después de desplegarla.
Instrucciones de configuración
1. Abre tu Google Sheet → Extensiones → Apps Script → pega el código → Guarda (Ctrl+S).
2. Implementar → Nueva implementación → Aplicación web → Ejecutar como: Yo → Quién tiene acceso: Cualquier persona → Implementar → Copia la URL.
3. Pega la URL en el campo de arriba.
4. GitHub Pages: sube index.html → Settings → Pages → Branch: main → Save.
Código para Google Apps Script: 
function doPost(e) {
  var sheet = SpreadsheetApp
    .openById('1Aaj7t2bNjOfuvHazjxI6DLXN6cHUEAFc')
    .getActiveSheet();
  var d = JSON.parse(e.postData.contents);
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['FECHA','HORA','ENTRADA/SALIDA','GUARDIA','GUÍA',
      'COMPAÑÍA','NOMBRE/APELLIDO','PLACA','DISCO','KILOMETRAJE',
      'COMBUSTIBLE','RETORNO/DESTINO','LLANTA EMERGENCIA',
      'EXTINTOR','KIT','BINES','SACOS','GAVETAS','OBSERVACIONES']);
  }
  sheet.appendRow([d.fecha,d.hora,d.tipo,d.guardia,d.guia,d.compania,
    d.nombre,d.placa,d.disco,d.km,d.comb,d.destino,
    d.llanta,d.extintor,d.kit,d.bines,d.sacos,d.gavetas,d.obs]);
  return ContentService
    .createTextOutput(JSON.stringify({status:'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}
function doGet(e) {
  var sheet = SpreadsheetApp
    .openById('1Aaj7t2bNjOfuvHazjxI6DLXN6cHUEAFc')
    .getActiveSheet();
  return ContentService
    .createTextOutput(JSON.stringify(sheet.getDataRange().getValues()))
    .setMimeType(ContentService.MimeType.JSON);
}
